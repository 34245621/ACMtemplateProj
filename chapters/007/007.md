priority_queue<node>q;

int bfs(int s) {
    node tmp;
    tmp.x = s;
    tmp.wei = 0;

                tmp.x = v.first;
                tmp.wei = v.second+u.wei;

        }
    }
}

### 7.3.4 Floyd

> description:\
> 多源最短路，适合邻接矩阵存图\
> 等效于进行n遍Bellman-Ford\
> 复杂度$O(n^3)$

```cpp
const int INF = 0x3f3f3f3f;
const int MAXN = 100;
int n, m;
int mp[MAXN][MAXN];
int floyd() {
    for (int k = 1; k <= n; ++k) {
        for (int i = 1; i <= n; ++i) {
            for (int j = 1; j <= n; ++j) {
                if(i!=j)mp[i][j] = min(mp[i][j], mp[i][k] + mp[k][j]);
            }
        }
    }
}
```

## 7.4. 最小生成树

### 7.4.1 Prim

> description:\
> 过程类似于Dijkstra，

```cpp
#include <queue>
#include <map>
map<int, map<int,int>> graph;

//返回最小生成树的边权和
int prim(int s) {
    int res = 0;
    map<int,int> dis;
    map<int,bool> visited;
    for (int i = 1; i <= n; ++i) {
        dis[i] = INF;
        visited[i] = false;
    }
    dis[s] = 0;
    visited[s] = true;
    while (!visited[n]) {
        int u = -1;
        for (int i = 1; i <= n; ++i) {
            if (!visited[i] && (u == -1 || dis[i] < dis[u])) {
                u = i;
            }
        }
        visited[u] = true;
        for (int v : graph[u]) {
            if (!visited[v.first] && v.second < dis[v.first]) {
                dis[v.first] = v.second;
            }
        }
    }
    for (int i = 1; i <= n; ++i) {
        res += dis[i];
    }
    return res;
}
```

### 7.4.2 Kruskal

> description:\
> 从所有边中从小到大选取最小生成树的边，
> 需要使用并查集
> 适合链式前向星存图
> 得改改代码

```cpp
#include <queue>
#include <map>

//链式前向星存图
map<int, int> fa;
map<int, vector<pair<int, int>>> graph;
void init(int n) {
    for (int i = 1; i <= n; ++i) {
        fa[i] = i;
    }
}
int find(int x) {
    if (fa[x] == x) {
        return x;
    }
    return find(fa[x]);
}
void merge(int x, int y) {
    int fx = find(x);
    int fy = find(y);
    if (fx != fy) {
        fa[fx] = fy;
    }
}
int kruskal() {
    int res = 0;
    priority_queue<pair<int, pair<int, int>>> q;
    for (auto &v : graph) {
        for (auto &e : v.second) {
            q.push({e.second, {e.first, v.first}});
        }
    }
    while (!q.empty()) {
        auto e = q.top();
        q.pop();
        int fx = find(e.second.first);
        int fy = find(e.second.second);
        if (fx != fy) {
            res += e.first;
            merge(fx, fy);
        }
    }
    return res;
}

```

## 7.6. 拓扑排序

> description:\
> 按照点的度数关系排序，
> 不断摘叶子

```cpp
#include <queue>
#include <map>
#include <vector>
map<int,map<int,int>>mp;
map<int,set<int>>in;
map<int,int>deg;
vector<int>topo;
void dfs(int x)
{
    for(auto &y:in[x])
    {
        deg[y]-=1;
        if(deg[y]==0)
        {
            dfs(y);
        }
    }
    topo.push_back(x);
}
void topo_sort()
{
    for(auto &x:deg)
    {
        if(x.second==0)
        {
            dfs(x.first);
        }
    }
}
```

## 7.9. 并查集

> description:\
> union-find算法，
> 使用路径压缩和按秩合并后复杂度为$O(α(n))$
> 十分高效

```cpp
#include <vector>
#include <map>
#include <queue>
#include <cmath>

struct UnionFind {
    vector<int> fa;
    vector<int> rank;
    int n;
    UnionFind(int n) {
        this->n = n;
        fa.resize(n + 1);
        rank.resize(n + 1);
        for (int i = 1; i <= n; ++i) {
            fa[i] = i;
            rank[i] = 1;
        }
    }
    ~UnionFind() {
        fa.clear();
        rank.clear();
    }
    int find(int x) {
        if (fa[x] != x) {
            fa[x] = find(fa[x]);
        }
        return fa[x];
    }
    void merge(int x, int y) {
        int fx = find(x);
        int fy = find(y);
        if (fx != fy) {
            if (rank[fx] < rank[fy]) {
                fa[fx] = fy;
            } else {
                fa[fy] = fx;
                if (rank[fx] == rank[fy]) {
                    ++rank[fx];
                }
            }
        }
    }
};
```
