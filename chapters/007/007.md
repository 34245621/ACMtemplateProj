> description:\

```cpp
map<int, set<int>> graph;
queue<int> q;
map<int,bool> visited;
int bfs(int s) {
    q.push(s);
    visited[s] = true;
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        for (int v : graph[u]) {
            if (!visited[v]) {
                q.push(v);
                visited[v] = true;
            }
        }
    }
}
```
